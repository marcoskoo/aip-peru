   157тЖТ   157тЖТ//
   158тЖТ   158тЖТ// buildMergedTree takes the static AIP skeleton and the list of sections
   159тЖТ   159тЖТ// fetched from /api/aip-sections, and produces a tree where:
   160тЖТ   160тЖТ//   - Static sections keep their hardcoded titles (but get updated title
   161тЖТ   161тЖТ//     if the DB has a newer one from an uploaded .md file)
   162тЖТ   162тЖТ//   - DB sections whose code is NOT in the static tree are appended:
   163тЖТ   163тЖТ//       * to an existing group (same part + subPart) if one exists
   164тЖТ   164тЖТ//       * to a new "[part] [subPart] - Secciones cargadas" group otherwise
   165тЖТ   165тЖТ//       * to a brand-new "[part] - Secciones cargadas" part if the part
   166тЖТ   166тЖТ//         itself doesn't exist (e.g. a custom part like "SUP")
   167тЖТ   167тЖТ// Each appended section is flagged with `uploaded: true` so the UI can
   168тЖТ   168тЖТ// show a distinct badge for content that came from a .md upload.
   169тЖТ   169тЖТ
   170тЖТ   170тЖТfunction buildMergedTree(dbSections: AipSectionSummary[]): PartNode[] {
   171тЖТ   171тЖТ  // Deep clone the static tree so we can mutate safely
   172тЖТ   172тЖТ  const tree: PartNode[] = STATIC_SECTION_TREE.map(p => ({
   173тЖТ   173тЖТ    ...p,
   174тЖТ   174тЖТ    groups: p.groups.map(g => ({
   175тЖТ   175тЖТ      ...g,
   176тЖТ   176тЖТ      sections: g.sections.map(s => ({ ...s })),
   177тЖТ   177тЖТ    })),
   178тЖТ   178тЖТ  }))
   179тЖТ   179тЖТ
   180тЖТ   180тЖТ  // Index existing sections by code for O(1) lookup
   181тЖТ   181тЖТ  const existingCodes = new Set<string>()
   182тЖТ   182тЖТ  for (const part of tree) {
   183тЖТ   183тЖТ    for (const group of part.groups) {
   184тЖТ   184тЖТ      for (const sec of group.sections) {
   185тЖТ   185тЖТ        existingCodes.add(sec.code)
   186тЖТ   186тЖТ      }
   187тЖТ   187тЖТ    }
   188тЖТ   188тЖТ  }
   189тЖТ   189тЖТ
   190тЖТ   190тЖТ  // Index parts and groups by their key for quick lookup
   191тЖТ   191тЖТ  const partIndex = new Map<string, PartNode>()
   192тЖТ   192тЖТ  for (const p of tree) partIndex.set(p.part, p)
   193тЖТ   193тЖТ
   194тЖТ   194тЖТ  for (const dbSec of dbSections) {
   195тЖТ   195тЖТ    // Skip sections already in the static tree (title is already set there,
   196тЖТ   196тЖТ    // but we update the title to the DB value in case an upload changed it)
   197тЖТ   197тЖТ    if (existingCodes.has(dbSec.sectionCode)) {
   198тЖТ   198тЖТ      for (const part of tree) {
   199тЖТ   199тЖТ        for (const group of part.groups) {
   200тЖТ   200тЖТ          for (const sec of group.sections) {
   201тЖТ   201тЖТ            if (sec.code === dbSec.sectionCode) {
   202тЖТ   202тЖТ              sec.title = dbSec.title
   203тЖТ   203тЖТ              sec.uploaded = !!dbSec.sourceFile
   204тЖТ   204тЖТ            }
   205тЖТ   205тЖТ          }
   206тЖТ   206тЖТ        }
   207тЖТ   207тЖТ      }
   208тЖТ   208тЖТ      continue
   209тЖТ   209тЖТ    }
   210тЖТ   210тЖТ
   211тЖТ   211тЖТ    // This is a new section from the DB (uploaded via .md) тАФ merge it in
   212тЖТ   212тЖТ    const partKey = (dbSec.part || 'GEN').toUpperCase()
   213тЖТ   213тЖТ    let part = partIndex.get(partKey)
   214тЖТ   214тЖТ    if (!part) {
   215тЖТ   215тЖТ      // Create a new part node for unknown parts (e.g. "SUP" for supplements)
   216тЖТ   216тЖТ      part = {
   217тЖТ   217тЖТ        part: partKey,
   218тЖТ   218тЖТ        label: `${partKey} - Secciones cargadas`,
   219тЖТ   219тЖТ        labelEn: 'Uploaded sections',
   220тЖТ   220тЖТ        icon: FileText,
   221тЖТ   221тЖТ        groups: [],
   222тЖТ   222тЖТ      }
   223тЖТ   223тЖТ      tree.push(part)
   224тЖТ   224тЖТ      partIndex.set(partKey, part)
   225тЖТ   225тЖТ    }
   226тЖТ   226тЖТ
   227тЖТ   227тЖТ    // Find or create the group (by subPart)
   228тЖТ   228тЖТ    const subPartKey = String(dbSec.subPart || '0')
   229тЖТ   229тЖТ    let group = part.groups.find(g => g.subPart === subPartKey)
   230тЖТ   230тЖТ    if (!group) {
   231тЖТ   231тЖТ      group = {
   232тЖТ   232тЖТ        subPart: subPartKey,
   233тЖТ   233тЖТ        label: `${partKey} ${subPartKey} - Secciones cargadas`,
   234тЖТ   234тЖТ        sections: [],
   235тЖТ   235тЖТ      }
   236тЖТ   236тЖТ      part.groups.push(group)
   237тЖТ   237тЖТ    }
   238тЖТ   238тЖТ
   239тЖТ   239тЖТ    group.sections.push({
   240тЖТ   240тЖТ      code: dbSec.sectionCode,
   241тЖТ   241тЖТ      title: dbSec.title,
   242тЖТ   242тЖТ      uploaded: true,
   243тЖТ   243тЖТ    })
   244тЖТ   244тЖТ  }
   245тЖТ   245тЖТ
   246тЖТ   246тЖТ  // Sort groups within each part by subPart (numeric-aware)
   247тЖТ   247тЖТ  for (const part of tree) {
   248тЖТ   248тЖТ    part.groups.sort((a, b) => {
   249тЖТ   249тЖТ      const na = parseFloat(a.subPart) || 0
   250тЖТ   250тЖТ      const nb = parseFloat(b.subPart) || 0
   251тЖТ   251тЖТ      return na - nb
   252тЖТ   252тЖТ    })
   253тЖТ   253тЖТ  }
   254тЖТ   254тЖТ
   255тЖТ   255тЖТ  return tree
   256тЖТ   256тЖТ}

   302тЖТ   302тЖТ
   303тЖТ   303тЖТ  // Build the merged tree (static skeleton + uploaded sections from DB)

   305тЖТ   305тЖТ
   306тЖТ   306тЖТ  // Count uploaded sections for the status badge

   308тЖТ   308тЖТ
   309тЖТ   309тЖТ  // Fetch section data
   310тЖТ   310тЖТ  const fetchSectionData = useCallback(async (code: string) => {

   315тЖТ   315тЖТ      setSectionData(d)

   317тЖТ   317тЖТ      setSectionData(null)
   318тЖТ   318тЖТ    } finally {
   319тЖТ   319тЖТ      setSectionLoading(false)
   320тЖТ   320тЖТ    }
   321тЖТ   321тЖТ  }, [])
   322тЖТ   322тЖТ
   323тЖТ   323тЖТ  useEffect(() => {
   324тЖТ   324тЖТ    if (selectedCode) {
   325тЖТ   325тЖТ      fetchSectionData(selectedCode)
   326тЖТ   326тЖТ    }
   327тЖТ   327тЖТ  }, [selectedCode, fetchSectionData])
   328тЖТ   328тЖТ
   329тЖТ   329тЖТ  const toggleGroup = (part: string) => {
   330тЖТ   330тЖТ    setExpandedGroups(prev => {
   331тЖТ   331тЖТ      const next = new Set(prev)
   332тЖТ   332тЖТ      if (next.has(part)) next.delete(part)
   333тЖТ   333тЖТ      else next.add(part)
