   161тЖТ   161тЖТ/** Great-circle distance in NM (haversine) */
   162тЖТ   162тЖТfunction gcDistNM(a: Coord, b: Coord): number {
   163тЖТ   163тЖТ  const R = 3440.065 // Earth radius in NM
   164тЖТ   164тЖТ  const dLat = ((b.lat - a.lat) * Math.PI) / 180
   165тЖТ   165тЖТ  const dLon = ((b.lon - a.lon) * Math.PI) / 180
   166тЖТ   166тЖТ  const lat1 = (a.lat * Math.PI) / 180
   167тЖТ   167тЖТ  const lat2 = (b.lat * Math.PI) / 180
   168тЖТ   168тЖТ  const h =
   169тЖТ   169тЖТ    Math.sin(dLat / 2) ** 2 +
   170тЖТ   170тЖТ    Math.cos(lat1) * Math.cos(lat2) * Math.sin(dLon / 2) ** 2
   171тЖТ   171тЖТ  return Math.round(R * 2 * Math.atan2(Math.sqrt(h), Math.sqrt(1 - h)))
   172тЖТ   172тЖТ}

   174тЖТ   174тЖТ/** Initial bearing from a to b (degrees true) */
   175тЖТ   175тЖТfunction bearingDeg(a: Coord, b: Coord): number {
   176тЖТ   176тЖТ  const dLon = ((b.lon - a.lon) * Math.PI) / 180
   177тЖТ   177тЖТ  const lat1 = (a.lat * Math.PI) / 180
   178тЖТ   178тЖТ  const lat2 = (b.lat * Math.PI) / 180
   179тЖТ   179тЖТ  const y = Math.sin(dLon) * Math.cos(lat2)
   180тЖТ   180тЖТ  const x =
   181тЖТ   181тЖТ    Math.cos(lat1) * Math.sin(lat2) -
   182тЖТ   182тЖТ    Math.sin(lat1) * Math.cos(lat2) * Math.cos(dLon)
   183тЖТ   183тЖТ  return Math.round(((((Math.atan2(y, x) * 180) / Math.PI) + 360) % 360))
   184тЖТ   184тЖТ}

   199тЖТ   199тЖТ/** Find nearest known point (airport/navaid/waypoint) within threshold degrees */
   200тЖТ   200тЖТfunction findNearestPoint(
   201тЖТ   201тЖТ  lat: number,
   202тЖТ   202тЖТ  lon: number,
   203тЖТ   203тЖТ  airports: PeruvianAirport[],
   204тЖТ   204тЖТ  navaids: Navaid[],
   205тЖТ   205тЖТ  waypoints: Waypoint[],
   206тЖТ   206тЖТ  threshold = 0.4
   207тЖТ   207тЖТ): RoutePoint | null {
   208тЖТ   208тЖТ  let nearest: RoutePoint | null = null
   209тЖТ   209тЖТ  let minDist = Infinity
   210тЖТ   210тЖТ  for (const ap of airports) {
   211тЖТ   211тЖТ    const d = Math.hypot(ap.lat - lat, ap.lng - lon)
   212тЖТ   212тЖТ    if (d < minDist && d < threshold) {
   213тЖТ   213тЖТ      minDist = d
   214тЖТ   214тЖТ      nearest = { id: ap.icao, name: ap.short || ap.city, type: "AIRPORT", lat: ap.lat, lon: ap.lng }
   215тЖТ   215тЖТ    }
   216тЖТ   216тЖТ  }
   217тЖТ   217тЖТ  for (const nv of navaids) {
   218тЖТ   218тЖТ    const d = Math.hypot(nv.lat - lat, nv.lon - lon)
   219тЖТ   219тЖТ    if (d < minDist && d < threshold) {
   220тЖТ   220тЖТ      minDist = d
   221тЖТ   221тЖТ      nearest = { id: nv.id, name: nv.name, type: "NAVAID", lat: nv.lat, lon: nv.lon }
   222тЖТ   222тЖТ    }
   223тЖТ   223тЖТ  }
   224тЖТ   224тЖТ  if (waypoints) {
   225тЖТ   225тЖТ    for (const wp of waypoints) {
   226тЖТ   226тЖТ      const d = Math.hypot(wp.lat - lat, wp.lon - lon)
   227тЖТ   227тЖТ      if (d < minDist && d < threshold) {
   228тЖТ   228тЖТ        minDist = d
   229тЖТ   229тЖТ        nearest = { id: wp.id, name: wp.name, type: "WAYPOINT", lat: wp.lat, lon: wp.lon }
   230тЖТ   230тЖТ      }
   231тЖТ   231тЖТ    }
   232тЖТ   232тЖТ  }
   233тЖТ   233тЖТ  return nearest
   234тЖТ   234тЖТ}
