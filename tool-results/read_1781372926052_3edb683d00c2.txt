    46тЖТ    46тЖТfunction dmsToDecimal(dms: string | null | undefined): number | null {
    47тЖТ    47тЖТ  if (!dms) return null
    48тЖТ    48тЖТ  // Parse formats like: 12┬░00'23.00"S, 077┬░07'07.00"W
    49тЖТ    49тЖТ  const match = dms.match(/(\d+)┬░(?:(\d+)['тА▓])?(?:(\d+(?:\.\d+)?)['тА│"])?\s*([NSEW])/i)
    50тЖТ    50тЖТ  if (!match) return null
    51тЖТ    51тЖТ  const degrees = parseFloat(match[1])
    52тЖТ    52тЖТ  const minutes = match[2] ? parseFloat(match[2]) : 0
    53тЖТ    53тЖТ  const seconds = match[3] ? parseFloat(match[3]) : 0
    54тЖТ    54тЖТ  const direction = match[4].toUpperCase()
    55тЖТ    55тЖТ  let decimal = degrees + minutes / 60 + seconds / 3600
    56тЖТ    56тЖТ  if (direction === 'S' || direction === 'W') decimal = -decimal
    57тЖТ    57тЖТ  return decimal
    58тЖТ    58тЖТ}

   497тЖТ   497тЖТfunction SunriseSunsetTab() {
   498тЖТ   498тЖТ  const [latitude, setLatitude] = useState("-12.0")
   499тЖТ   499тЖТ  const [longitude, setLongitude] = useState("-77.0")
   500тЖТ   500тЖТ  const [date, setDate] = useState(new Date().toISOString().split("T")[0])
   501тЖТ   501тЖТ
   502тЖТ   502тЖТ  const handleAirportSelect = useCallback((airport: { icaoCode: string; latitude: number; longitude: number }) => {
   503тЖТ   503тЖТ    setLatitude(String(Math.round(airport.latitude * 100) / 100))
   504тЖТ   504тЖТ    setLongitude(String(Math.round(airport.longitude * 100) / 100))
   505тЖТ   505тЖТ  }, [])
   506тЖТ   506тЖТ
   507тЖТ   507тЖТ  const result = useMemo(() => {
   508тЖТ   508тЖТ    const lat = parseFloat(latitude)
   509тЖТ   509тЖТ    const lon = parseFloat(longitude)
   510тЖТ   510тЖТ    const d = date ? new Date(date) : new Date()
   511тЖТ   511тЖТ    if (isNaN(lat) || isNaN(lon)) return null
   512тЖТ   512тЖТ
   513тЖТ   513тЖТ    // Simplified NOAA Solar Calculator
   514тЖТ   514тЖТ    const dayOfYear = Math.floor((d.getTime() - new Date(d.getFullYear(), 0, 0).getTime()) / 86400000)
   515тЖТ   515тЖТ
   516тЖТ   516тЖТ    // Solar declination (degrees)
   517тЖТ   517тЖТ    const declination = -23.45 * Math.cos((2 * Math.PI / 365) * (dayOfYear + 10))
   518тЖТ   518тЖТ    const decRad = (declination * Math.PI) / 180
   519тЖТ   519тЖТ    const latRad = (lat * Math.PI) / 180
   520тЖТ   520тЖТ
   521тЖТ   521тЖТ    // Hour angle for sunrise/sunset (degrees)
   522тЖТ   522тЖТ    const cosHourAngle = (Math.sin((-0.833 * Math.PI) / 180) - Math.sin(latRad) * Math.sin(decRad)) /
   523тЖТ   523тЖТ      (Math.cos(latRad) * Math.cos(decRad))
   524тЖТ   524тЖТ
   525тЖТ   525тЖТ    // Civil twilight hour angle
   526тЖТ   526тЖТ    const cosCivilTwilight = (Math.sin((-6 * Math.PI) / 180) - Math.sin(latRad) * Math.sin(decRad)) /
   527тЖТ   527тЖТ      (Math.cos(latRad) * Math.cos(decRad))
   528тЖТ   528тЖТ
   529тЖТ   529тЖТ    if (cosHourAngle > 1 || cosHourAngle < -1) {
   530тЖТ   530тЖТ      return {
   531тЖТ   531тЖТ        isPolar: true,
   532тЖТ   532тЖТ        message: cosHourAngle > 1 ? "Noche polar тАФ sin amanecer" : "D├нa polar тАФ sin atardecer",
   533тЖТ   533тЖТ      }
   534тЖТ   534тЖТ    }
   535тЖТ   535тЖТ
   536тЖТ   536тЖТ    const hourAngle = (Math.acos(cosHourAngle) * 180) / Math.PI
   537тЖТ   537тЖТ    const civilTwilightAngle = (Math.acos(Math.min(1, Math.max(-1, cosCivilTwilight))) * 180) / Math.PI
   538тЖТ   538тЖТ
   539тЖТ   539тЖТ    // Equation of time (minutes) тАФ simplified
   540тЖТ   540тЖТ    const B = ((2 * Math.PI) / 365) * (dayOfYear - 81)
   541тЖТ   541тЖТ    const eotMin = 9.87 * Math.sin(2 * B) - 7.53 * Math.cos(B) - 1.5 * Math.sin(B)
   542тЖТ   542тЖТ
   543тЖТ   543тЖТ    // Solar noon (minutes from midnight UTC)
   544тЖТ   544тЖТ    const solarNoonUTC = 720 - 4 * lon - eotMin
   545тЖТ   545тЖТ
   546тЖТ   546тЖТ    // Sunrise/sunset UTC (minutes)

   548тЖТ   548тЖТ    const sunsetUTC = solarNoonUTC + hourAngle * 4
   549тЖТ   549тЖТ    const civilDawnUTC = solarNoonUTC - civilTwilightAngle * 4
   550тЖТ   550тЖТ    const civilDuskUTC = solarNoonUTC + civilTwilightAngle * 4
   551тЖТ   551тЖТ
   552тЖТ   552тЖТ    // Daylight duration

   554тЖТ   554тЖТ    const daylightH = Math.floor(daylightMin / 60)
   555тЖТ   555тЖТ    const daylightM = Math.round(daylightMin % 60)
   556тЖТ   556тЖТ

   559тЖТ   559тЖТ      const m = Math.round(((minutes % 1440) + 1440) % 1440 % 60)
   560тЖТ   560тЖТ      return `${String(h).padStart(2, "0")}:${String(m).padStart(2, "3")} UTC`
   561тЖТ   561тЖТ    }
   562тЖТ   562тЖТ
   563тЖТ   563тЖТ    return {
   564тЖТ   564тЖТ      sunrise: formatUTC(sunriseUTC),
   565тЖТ   565тЖТ      sunset: formatUTC(sunsetUTC),
   566тЖТ   566тЖТ      civilDawn: formatUTC(civilDawnUTC),
   567тЖТ   567тЖТ      civilDusk: formatUTC(civilDuskUTC),
   568тЖТ   568тЖТ      daylight: `${daylightH}h ${daylightM}m`,
   569тЖТ   569тЖТ      declination: Math.round(declination * 10) / 10,
   570тЖТ   570тЖТ      isPolar: false,
   571тЖТ   571тЖТ    }
   572тЖТ   572тЖТ  }, [latitude, longitude, date])
   573тЖТ   573тЖТ
